Bonus: 2.1.2.1 Scale functions

Scales are mainly used for transforming data values to visual variables such as position, length and colour.

 			    
			

// http://d3indepth.com/scales/

// Dimensions
var h = 200,
	w = 500;
var margin = {top: 0, right: 10, bottom: 0, left: 20};
var width =  w - margin.left - margin.right,
	height = h - margin.top -  margin.bottom;


// DATA

var data = [ 0, 2, 3, 5, 7.5, 9, 10 ];
var extent = d3.extent(data);

// SCALE
var myScale = d3.scaleLinear()
	.domain(extent)
	.range([0, width]);

// DRAWING AREA	
var mySVG = d3.select("#viz")
.append('svg:svg')
.attr("width", w)
.attr("height", h)
.attr("preserveAspectRatio", "xMinYMin meet")
.attr("viewBox", "0 0 500 200");;

// Create a group and append it to the drawing area.
var g = mySVG.append('svg:g');

// COLORS
var colors = d3.scaleOrdinal(d3.schemeCategory10);
	

g.selectAll('circle')
.data(data)
.enter()
.append('circle')
.attr('r', 3)
.attr('fill', function(d, i){ return colors(i)})
.attr('cx', function(d) {
	return myScale(d);
});	

g.selectAll('text')
.data(data)
.enter()
.append('text')
.attr('x', function(d) {
	return myScale(d);
})
.attr('y', -8)
.text(function(d) {
	return d;
})
.style("text-anchor", "middle");

g.attr("transform", "translate(" + margin.left + "," + h/2 + ")");
