Bonus: 2.2.3 Your First Animated D3 Chart

Click the bars.

Each bar will have an initial length, and when you click it, it will expand to its final length. Remember, the barsâ€™ lengths are actually the rectangles widths, since you are drawing horizontal bars here.

 			    
			

var h = 500,
	w = 500;
var margin = {top: 10, right: 10, bottom: 20, left: 10};
var width =  w - margin.left - margin.right,
	height = h - margin.top -  margin.bottom;

var gdp_data = [
  [10,15],
  [15,25],
  [25,30],
  [10,25]
];
// Calculate the spacings between bars.
var spacing  = height / (gdp_data.length + 1);

// Calculate the heightist gdp value, and create a scale using it.
var max_gdp = d3.max(gdp_data, function(array) {
	return d3.max(array);
});
var min_gdp = d3.min(gdp_data, function(array) {
	return d3.min(array);
});


// SCALES
var xScale = d3.scaleLinear()
.domain([0, max_gdp])
.range([0, width]);

// Scale for bars vertical positions.
var yScale  = d3.scaleLinear()
.domain([0, gdp_data.length + 1])
.range([0, height]);


// AXIS
var xAxis = d3.axisBottom(xScale).tickSize(-height).tickPadding(5);



// DRAW
// Draw the SVG

var mySVG = d3.select("#viz")
	.append("svg")
	.attr("width", w)
	.attr("height", h)
	.attr("preserveAspectRatio", "xMinYMin meet")
	.attr("viewBox", "0 0 500 500")
	.append("g")
	.attr("transform", "translate(" + margin.left + "," + margin.top + ")");

// Draw the Axis	
 mySVG.append("g")
	.attr("id", "xAxisG")
	.call(xAxis);

d3.selectAll("#xAxisG")
	.attr("transform","translate(0," + height + ")");

// Draw the GDP bars

mySVG.selectAll("rect")
	.data(gdp_data)
	.enter()
	.append("rect")
	.attr("y", function (d, i) {
		return yScale(i+.5);
	})
	.attr("x", 0)
	.attr("width", function (d, i) {
 		return xScale(d[0]);
	})
	.attr("height", "60")
	.style("fill", "grey")
	.style("stroke", "none")
	.style("stroke-width", 0)
	
	.on('mouseenter', function(d) {
    	d3.select(this).style("cursor", "pointer")
    })
	.on('click', function(d, i){
		d3.select(this)
		.style("stroke-width", 1)
		.style("fill", "blue")
		.transition()
		.delay(100)
		.duration(500)
		.attr("width", function (d, i) {
  			return xScale(d[1]);
 		});
	});
